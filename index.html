package coordenadas;
import java.util.Scanner;
public class Coordenadas {// inicio coordenadas
public static void main(String[] args) {// inicio main
// NOTA EL LADO CA ES LA BASE, LADO BC ES LA ALTURA Y LADO AB ES LA HIPOTENUSA
Scanner scanner = new Scanner(System.in);
System.out.println("NOTA EL LADO CA ES LA BASE, LADO BC ES LA ALTURA Y LADO AB ES LA HIPOTENUSA");
// Leer datos de las cooordenadas
System.out.println("Ingrese la coordenada x del vértice A: ");
int x1 = scanner.nextInt();
System.out.println("Ingrese la coordenada y del vértice A: ");
int y1 = scanner.nextInt();
System.out.println("Ingrese la coordenada x del vértice B: ");
int x2 = scanner.nextInt();
System.out.println("Ingrese la coordenada y del vértice B: ");
int y2 = scanner.nextInt();
System.out.println("Ingrese la coordenada x del vértice C: ");
int x3 = scanner.nextInt();
System.out.println("Ingrese la coordenada y del vértice C: ");
int y3 = scanner.nextInt();
// Calcular cada lado del triángulo
double ladoAB = Math.sqrt((x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1));
double ladoBC = Math.sqrt((x3 - x2) * (x3 - x2) + (y3 - y2) * (y3 - y2));
double ladoCA = Math.sqrt((x1 - x3) * (x1 - x3) + (y1 - y3) * (y1 - y3));
// NOTA EL LADO CA ES LA BASE, LADO BC ES LA ALTURA Y LADO AB ES LA HIPOTENUSA
// tipo de triángulo
String tipoTriangulo;
double area;
if (ladoAB == ladoBC && ladoBC == ladoCA) {
tipoTriangulo = "Equilátero";
// Calcular el área del triángulo
double a = ladoCA / 2;
double altura = Math.sqrt((ladoAB * ladoAB) - (a * a));
area = (ladoCA * altura) / 2;
} else if (ladoAB == ladoBC || ladoAB == ladoCA || ladoBC == ladoCA) {
tipoTriangulo = "Isósceles";
// Calcular el área del triángulo
double a = ladoCA / 2;
double altura = Math.sqrt((ladoAB * ladoAB) - (a * a));
area = (ladoCA * altura) / 2;
} else {
tipoTriangulo = "Escaleno";
// Calcular el área del triángulo
area = (ladoCA * ladoBC) / 2;
}
// Calcular el perímetro del triángulo
double perimetro = ladoAB + ladoBC + ladoCA;
// NOTA EL LADO CA ES LA BASE, LADO BC ES LA ALTURA Y LADO AB ES LA HIPOTENUSA
// Calcular los ángulos del triángulo usando la ley de los cosenos
double ladoA = ladoCA;
double ladoB = ladoBC;
double ladoC = ladoAB;
double anguloA = Math.toDegrees(Math.acos((ladoB * ladoB + ladoC * ladoC - ladoA * ladoA) / (2 * ladoB * ladoC)));
double anguloB = Math.toDegrees(Math.acos((ladoA * ladoA + ladoC * ladoC - ladoB * ladoB) / (2 * ladoA * ladoC)));
double anguloC = 180 - anguloA - anguloB;
// Imprimir resultados
System.out.println("-----------------------------------------------------------------------");
System.out.println("Las coordenadas del vertice A ingresadas fueron: x= " + x1 + " y= " + y1);
System.out.println("Las coordenadas del vertice B ingresadas fueron: x= " + x2 + " y= " + y2);
System.out.println("Las coordenadas del vertice C ingresadas fueron: x= " + x3 + " y= " + y3);
System.out.println("La distancia entre los vertices(Puntos) AB (hipotenusa)= " + ladoAB);
System.out.println("La distancia entre los vertices(Puntos) BC (Altura)= " + ladoBC);
System.out.println("La distancia entre los vertices(Puntos) AC3= " + ladoCA);
System.out.println("Esto representa que es un triangulo " + tipoTriangulo);
System.out.println("El perimetro del triangulo es de = " + perimetro);
System.out.println("El area del triangulo es de = " + area);
System.out.println("El angulos a= " + anguloA);
System.out.println("El angulos b= " + anguloB);
System.out.println("El angulos c= " + anguloC);
}//fin main }// fin coorddenadas
